 * Clase principal para crear nuevas clases Java desde la CLI.
 */

public class CreateClassCommand implements Callable<Integer> {

     * Nombre del paquete de la clase.
     */

    private String packageName;

    public Integer call() throws IOException {
        // lógica de generación con JavaPoet
        return 0;
    }
}
